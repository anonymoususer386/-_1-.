#include <iostream>
using namespace std;

int main() {
    double x, y; // координати точки для перевірки

    cout << "Enter x: ";
    cin >> x;
    cout << "Enter y: ";
    cin >> y;

    // Прямокутник: x ∈ [0,2], y ∈ [0,1]
    if (x >= 0 && x <= 2 && y >= 0 && y <= 1) {
        cout << "Point (" << x << ", " << y << ") is inside the rectangle." << endl;
    } else {
        cout << "Point (" << x << ", " << y << ") is outside the rectangle." << endl;
    }

    return 0;
}
